    let userAnswers = { ch1: {}, ch2: {}, ch3: {} };


    // --- INITIALIZATION ---

    function renderQuestions() {

        ['ch1', 'ch2', 'ch3'].forEach(ch => {

            const container = document.getElementById(`${ch}-questions`);

            container.innerHTML = ''; // Clear existing

            

            quizData[ch].questions.forEach((qObj, index) => {

                const qDiv = document.createElement('div');

                qDiv.className = 'question-block';

                

                let optionsHTML = '';

                qObj.options.forEach((opt, optIndex) => {

                    optionsHTML += `

                        <label class="option-label">

                            <input type="radio" name="${ch}-q${index}" value="${optIndex}" 

                            onchange="saveAnswer('${ch}', ${index}, ${optIndex})">

                            ${opt}

                        </label>

                    `;

                });


                qDiv.innerHTML = `

                    <div class="question-text">Q${index + 1}. ${qObj.q}</div>

                    <div class="options-grid">${optionsHTML}</div>

                `;

                container.appendChild(qDiv);

            });

        });

    }


    // --- NAVIGATION ---

    function switchTab(targetId) {

        // Update Buttons

        document.querySelectorAll('.nav-btn').forEach(btn => btn.classList.remove('active'));

        event.target.classList.add('active');


        // Update Screens

        document.querySelectorAll('.screen').forEach(screen => screen.classList.remove('active'));

        document.getElementById(targetId).classList.add('active');

    }


    // --- LOGIC ---

    function saveAnswer(chapter, questionIndex, optionIndex) {

        userAnswers[chapter][questionIndex] = optionIndex;

    }


    function submitQuiz(chapter) {

        const totalQ = quizData[chapter].questions.length;

        const answeredQ = Object.keys(userAnswers[chapter]).length;


        if (answeredQ < totalQ) {

            alert(`Please answer all questions before submitting. You have answered ${answeredQ} out of ${totalQ}.`);

            return;

        }


        generateReport(chapter);

        

        // Hide tabs, show report

        document.getElementById('nav-tabs').style.display = 'none';

        document.querySelectorAll('.screen').forEach(screen => screen.classList.remove('active'));

        document.getElementById('report-screen').classList.add('active');

    }


    function generateReport(chapter) {

        const reportContainer = document.getElementById('report-content');

        const data = quizData[chapter];

        let score = 0;

        let reviewHTML = '';


        data.questions.forEach((qObj, index) => {

            const userAns = userAnswers[chapter][index];

            const isCorrect = userAns === qObj.ans;

            if (isCorrect) score++;


            reviewHTML += `

                <div class="review-item ${isCorrect ? 'correct' : 'incorrect'}">

                    <div class="question-text">Q${index + 1}. ${qObj.q}</div>

                    <p>Your Answer: <span class="${isCorrect ? 'correct-text' : 'incorrect-text'}">

                        ${qObj.options[userAns]}

                    </span></p>

                    ${!isCorrect ? `<p>Correct Answer: <span class="correct-text">${qObj.options[qObj.ans]}</span></p>` : ''}

                    <div class="explanation"><strong>Learning Note:</strong> ${qObj.exp}</div>

                </div>

            `;

        });


        reportContainer.innerHTML = `

            <div class="report-header">

                <h2>Performance Report: ${data.title}</h2>

                <p>Date: ${new Date().toLocaleDateString()}</p>

            </div>

            <div class="score-card">

                <h3>Total Marks Obtained</h3>

                <h2>${score} / ${data.questions.length}</h2>

            </div>

            <h3>Detailed Review</h3>

            ${reviewHTML}

        `;

    }


    function resetQuizzes() {

        userAnswers = { ch1: {}, ch2: {}, ch3: {} };

        document.querySelectorAll('input[type="radio"]').forEach(radio => radio.checked = false);

        

        document.getElementById('nav-tabs').style.display = 'flex';

        

        // Switch back to Ch1

        document.querySelectorAll('.nav-btn')[0].click();

    }


    function exportPDF() {

        const element = document.getElementById('report-content');

        const opt = {

            margin:       10,

            filename:     'Physics_Quiz_Report.pdf',

            image:        { type: 'jpeg', quality: 0.98 },

            html2canvas:  { scale: 2 },

            jsPDF:        { unit: 'mm', format: 'a4', orientation: 'portrait' }

        };

        html2pdf().set(opt).from(element).save();

    }
